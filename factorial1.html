Factorial1

Sursa: pbinfo

Cerinţă

Să se scrie un program care citește numărul natural n și determină valoarea lui n! = 1 * 2 * 3 * ... * n.


Date de intrare

Programul citește de la tastatură numărul n.


Date de ieşire

Programul afișează pe prima linie a ecranului valoarea lui n!


Restricţii

  • 0 ≤ n ≤ 1000
  • Intrare/ ieşire: tastatură/ ecran
  • Timp maxim de execuţie/ test: 0.1s
  • Memorie totala disponibilă/ stivă: 64MB/ 8MB

Exemple

Intrare Ieşire
4 24
25 15511210043330985984000000

Implementare

	#include "iostream"

	using namespace std;

	const int NMAX = 1005;

	typedef int bigInt[NMAX];

	void atribIntToBigInt(bigInt A, int x) {
    	   // lui A (număr mare) i se atribuie valoarea lui x
    	   A[0] = 0;
    	   while (x) {
        	A[++ A[0]] = x % 10;
        	x /= 10;
    	   }
	}

	void multiplyBigIntByInt(bigInt A, int x) {
    	   // înmulţirea unui număr mare cu un număr mic
    	   // A = A * x
    	   int t = 0; // transportul
    	   for (int i = 1; i <= A[0]; ++ i) {
        	A[i] = A[i] * x + t;
        	t = A[i] / 10;
        	A[i] %= 10;
    	   }
    	   while (t) {
        	A[++ A[0]] = t % 10;
        	t /= 10;
    	   }
	}

	void solve(int n, bigInt A) {
    	   for (int i = 1; i <= n; ++ i)
        	multiplyBigIntByInt(A, i);
	}

	void printBigInt(bigInt A) {
    	   // afişarea unui număr mare
    	   for (int i = A[0]; i; -- i)
        	cout << A[i];
    	   cout << "\n";
	}

	int main() {
    	   int n;
    	   bigInt A;
    	   cin >> n;
    	   atribIntToBigInt(A, 1);
    	   solve(n, A);
    	   printBigInt(A);
    	   return 0;
	}
